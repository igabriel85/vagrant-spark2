# Example Spark submit

spark-submit --class org.apache.spark.examples.SparkPi --master spark://ubuntu-jammy:7077 /usr/local/spark/examples/jars/spark-examples_2.12-3.5.7.jar 100


spark-submit --master spark://ubuntu-jammy:7077 --executor-memory 1g  --driver-class-path /usr/local/spark/examples/src/main/python/ml/gradient_boosted_tree_classifier_example.py /usr/local/spark/examples/src/main/python/ml/gradient_boosted_tree_classifier_example.py

# Spark master and worker have to be started in order to execute Spark workloads.

./start-master.sh
./start-worker.sh spark://ubuntu-jammy:7077
or
./start-slave.sh spark://node1:7077


Note: Spark master is spark://ubuntu-jammy:7077 for single VM deployment and spark://node1:7077 for multi node deployment
